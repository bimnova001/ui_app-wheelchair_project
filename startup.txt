sudo nano /etc/systemd/system/burger.service

[Unit]
Description=Run burger robot nodes on boot
After=network.target

[Service]
Type=simple
WorkingDirectory=/home/pi/ui_app-wheelchair_project
ExecStart=/usr/bin/python3 /home/pi/ui_app-wheelchair_project/burger.py
Restart=always
User=pi

[Install]
WantedBy=multi-user.target


sudo systemctl daemon-reload
sudo systemctl enable burger.service
sudo systemctl start burger.service
sudo systemctl status burger.service
